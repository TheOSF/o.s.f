272 inline int32_t ConvertUtf16ToUtf8( int8_t* dst, int32_t dst_size, const int16_t* src )
273 {
274  int32_t cnt = 0;
275  const int16_t* wp = src;
276  int8_t* cp = dst;
277 
278  if (dst_size == 0) return 0;
279 
280  dst_size -= 3;
281 
282  for (cnt = 0; cnt < dst_size; )
283  {
284  int16_t wc = *wp++;
285  if (wc == 0)
286  {
287  break;
288  }
289  if ((wc & ~0x7f) == 0)
290  {
291  *cp++ = wc & 0x7f;
292  cnt += 1;
293  } else if ((wc & ~0x7ff) == 0)
294  {
295  *cp++ = ((wc >> 6) & 0x1f) | 0xc0;
296  *cp++ = ((wc) & 0x3f) | 0x80;
297  cnt += 2;
298  } else {
299  *cp++ = ((wc >> 12) & 0xf) | 0xe0;
300  *cp++ = ((wc >> 6) & 0x3f) | 0x80;
301  *cp++ = ((wc) & 0x3f) | 0x80;
302  cnt += 3;
303  }
304  }
305  *cp = '\0';
306  return cnt;
307 }

316 inline int32_t ConvertUtf8ToUtf16( int16_t* dst, int32_t dst_size, const int8_t* src )
317 {
318  int32_t i, code;
319  int8_t c0, c1, c2;
320 
321  if (dst_size == 0) return 0;
322 
323  dst_size -= 1;
324 
325  for (i = 0; i < dst_size; i++)
326  {
327  int16_t wc;
328 
329  c0 = *src++;
330  if (c0 == '\0')
331  {
332  break;
333  }
334  // UTF8からUTF16に変換
335  code = (uint8_t)c0 >> 4;
336  if (code <= 7)
337  {
338  // 8bit文字
339  wc = c0;
340  }
341  else if (code >= 12 && code <= 13)
342  {
343  // 16bit文字
344  c1 = *src++;
345  wc = ((c0 & 0x1F) << 6) | (c1 & 0x3F);
346  }
347  else if (code == 14)
348  {
349  // 24bit文字
350  c1 = *src++;
351  c2 = *src++;
352  wc = ((c0 & 0x0F) << 12) | ((c1 & 0x3F) << 6) | (c2 & 0x3F);
353  }
354  else
355  {
356  continue;
357  }
358  dst[i] = wc;
359  }
360  dst[i] = 0;
361  return i;
362 }
